            function dodawanie(a, b) {
                return a + b;
            }
            function odejmowanie(a, b) {
                return a - b;
            }
            function mnozenie(a, b) {
                return a * b;
            }
            function dzielenie(a, b) {
                return b !== 0 ? (a / b) : "Nie można dzielić przez zero!";
            }
            
            let x = 10;
            let y = 5;
            let wynik = `
                Dodawanie: ${x} + ${y} = ${dodawanie(x, y)}<br>
                Odejmowanie: ${x} - ${y} = ${odejmowanie(x, y)}<br>
                Mnożenie: ${x} * ${y} = ${mnozenie(x, y)}<br>
                Dzielenie: ${x} / ${y} = ${dzielenie(x, y)}<br>
            `;
            document.getElementById("wynik").innerHTML = wynik;

            function sprawdzOdpowiedz() {
                let odpowiedz = document.getElementById("quiz").value;
                let poprawna = 144;
                if (parseInt(odpowiedz) === poprawna) {
                    document.getElementById("odpowiedz").innerHTML = "Brawo! To poprawna odpowiedź!";
                } else {
                    document.getElementById("odpowiedz").innerHTML = "Niestety, spróbuj ponownie.";
                }
